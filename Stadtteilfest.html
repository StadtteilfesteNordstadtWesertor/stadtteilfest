      function renderLanguageOptions() {

        els.languageSelect.innerHTML = "";

        languageOptions.forEach((option) => {

          const op = document.createElement("option");

          op.value = option.code;

          op.textContent = option.label;

          els.languageSelect.appendChild(op);

        });

      }


      function applyLanguage(language) {

        const t = translations[language] || translations.de;

        document.documentElement.lang = language;

        document.documentElement.dir = language === "ar" ? "rtl" : "ltr";


        els.languageLabel.textContent = t.languageLabel;

        els.titleTop.textContent = t.titleTop;

        els.titleMain.textContent = t.titleMain;

        els.titlePlace.textContent = t.titlePlace;

        els.titleBottom.textContent = t.titleBottom;

        els.intro.textContent = t.intro;

        els.stripText.textContent = t.stripText;

        els.dateValue.textContent = t.dateValue;

        els.timeValue.textContent = t.timeValue;

        els.locationValue.textContent = t.locationValue;

        els.footerNote.textContent = t.footerNote;


        els.contentCopy.innerHTML = "";

        t.paragraphs.forEach((text) => {

          const p = document.createElement("p");

          p.textContent = text;

          els.contentCopy.appendChild(p);

        });


        els.languageSelect.value = language;

        const params = new URLSearchParams(window.location.search);

        params.set("lang", language);

        const newUrl = `${window.location.pathname}?${params.toString()}`;

        history.replaceState({}, "", newUrl);

      }


      renderLanguageOptions();


      const initialLanguage = (() => {

        const params = new URLSearchParams(window.location.search);

        const langFromUrl = params.get("lang");

        if (langFromUrl && translations[langFromUrl]) return langFromUrl;

        const browserLang = navigator.language?.slice(0, 2).toLowerCase();

        if (browserLang && translations[browserLang]) return browserLang;

        return "de";

      })();


      applyLanguage(initialLanguage);

      els.languageSelect.addEventListener("change", (event) => applyLanguage(event.target.value));
